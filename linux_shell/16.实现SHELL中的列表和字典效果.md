# vim test.sh
#!/bin/bash
family=(     #  也可以这样定义  family=(aaa bbb ccc)
aaa
bbb
ccc
)

for name in ${family[*]}   #第一种方法，常规for循环取值的方式
do
echo $name
done
echo "================="
for ((name=0;name<${#family[*]};name++))   #第二种方法
do
echo ${family[name]}
done

grep -w 是用来匹配整词的选项。具体来说，-w 选项确保 grep 只匹配完整的单词，而不是单词的一部分。区分大小写

# 正则区别

^shanghai.\*?：匹配从字符串开头到 shanghai 开头之后的任意少量字符，没有强制匹配到字符串结尾。
^shanghai.\*?$：匹配从字符串开头到 shanghai 开头之后的任意少量字符，但必须一直匹配到字符串的结尾。
.*?是为了匹配更少的 字符来满足正则的要求，那就没有比较加$了，$要求匹配到 数据的末尾
