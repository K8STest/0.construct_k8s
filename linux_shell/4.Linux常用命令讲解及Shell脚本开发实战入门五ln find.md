find
#查询/mnt目录下所有.log的文件每个的大小
# 注：xargs组合多个命令的一个工具

echo '1.log'|du -sh

find /opt/linux-shell/ -type f -name "*.log"| du -sh
# 管道符号 | 将 find 命令的输出传递给 du 命令，这样 du 命令就会对这些 .log 文件计算总的磁盘使用情况并以人类可读的格式显示。

# xargs：从标准输入读取项目，并将其作为参数传递给后面的命令。在这里，它会将 find 命令找到的每个 .log 文件路径传递给 du 命令，一起交给du命令
find /opt/linux-shell/ -type f -name "*.log"|xargs du -sh
du -sh /opt/linux-shell/test/1.log /opt/linux-shell/logs/1.log /opt/linux-shell/1.log

# 复制/mnt目录下所有的.log后缀的文件到./logs下, -exec：find 命令的一个选项，表示对找到的每个文件执行指定的命令, \;结束 -exec 选项的命令。
find /opt/linux-shell -type f -name "*.log" -exec cp -a {} ./logs \;


具体流程
假设在 /opt/linux-shell 目录下有以下 .log 文件：

/opt/linux-shell/file1.log
/opt/linux-shell/subdir/file2.log
执行上述命令后，find 命令会找到这些文件，并对每个文件执行 cp -a 命令，相当于依次执行以下命令：

cp -a /opt/linux-shell/file1.log ./logs
cp -a /opt/linux-shell/subdir/file2.log ./logs



# find过滤(find排除)文件和目录，这个实战没得到很好效果
#注： /mnt目录内  过滤test目录内所有文件   并同时   过滤filename.txt这个文件 ，-o：逻辑或运算符，表示如果前面的条件不满足，继续检查后面的条件
# -o：逻辑或运算符，用于组合多个条件。如果前一个条件不满足，则检查后一个条件。
# -print参数加与不加有很大差别，加上后，逻辑或的条件起到过滤作用，链式的效果；如果不加，有一个条件满足，就会输出文件信息
find /opt/linux-shell/ -type f -path "/opt/linux-shell/logs/*" -o -path "/opt/linux-shell/test/*" -o -type f -name "*.log" -print
find /opt/linux-shell/ \( -type f -path "/opt/linux-shell/logs/*" \) -o \( -type f -path "/opt/linux-shell/test/*" \) -o \( -type f -name "*.log" \) -print

find /opt/linux-shell/ -type f -path "/opt/linux-shell/logs/*" -o -type f -name "*.log" -print
find /opt/linux-shell/ \( -type f -path "/opt/linux-shell/logs/*" \)  -print



find ./ -type f -not -name "1.log"


# 比较下面两个打包的区别
# xargs 统一执行打包成一个，keep珠文件夹结构进行打包， exec 是对文件分别打包

find /opt/linux-shell/ \( -type f -path "/opt/linux-shell/logs/*" \) -o \( -type f -name "*.log" \) -print|xargs tar zcvf allLogs.tar.gz

find /opt/linux-shell/ \( -type f -path "/opt/linux-shell/logs/*" \) -o \( -type f -name "*.log" \) -print -exec tar zcvf {}.tar.gz {} \;
