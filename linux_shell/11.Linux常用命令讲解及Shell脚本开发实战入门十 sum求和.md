计算两个数的和

```
#!/bin/bash

checkNum(){
  local num=$1  # local 只对 这个方法内有效
  expr $num + 1 &>/dev/null  #
  if [[ $? -ne 0 ]];then  # $?判断上一个命令执行结果是否正确
    echo "$num was not a number."
    exit 1
  fi
}

read -p "pleat input first number: " num1
checkNum $num1
read -p "pleat input second number: " num2
checkNum $num2


sum=$(( num1 + num2 ))
echo "sum: $sum"

```

现代写法

```
#!/bin/bash

checkNum() {
  local num=$1  # local 只对 这个方法内有效
  if ! [[ "$num" =~ ^-?[0-9]+$ ]]; then  # 使用正则表达式检查数字
    echo "$num was not a number."
    exit 1
  fi
}

read -p "Please input first number: " num1
checkNum "$num1"
read -p "Please input second number: " num2
checkNum "$num2"

sum=$(( num1 + num2 ))  # 使用算术扩展计算
echo "Sum: $sum"
```
