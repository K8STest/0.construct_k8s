看到网上某某新闻又说谁谁不小心删除了一些重要文件，咱也不知道真假，但对于 Linux 发行版系统而言，如果操作不熟练、不严谨，是很容易造成误删除的情况出现的，那么这个情况下，如果我们能实现像 Windowns 系统下回收站的功能，那不是就减少出现误删除的情况了么。好了，话不多说，我们直接来看源码，同时大家可以关注博哥同名的视频教程，对着教程来理解下面有脚本源码：

```
#!/bin/bash
# $#脚本后参数数量， $1 第一个参数
if [[ $# -eq 0 ]];then
  echo "Usage: bash $0 files or dirs"
  exit 1
fi

function Help(){
  echo "Do you want to continue, input enter to next."
  read -p "" aaaaaaaaaaa
}

main(){
  mytemp=$1
  D=/temp/$(date +%Y%m%d%H%M%S_hsz)    # date +%Y%m%d%H%M%S_hsz 日期命令
  [ ! -d $D ] && mkdir -p $D

  # [ 和 ] 是 shell 中用于条件测试的命令，也称为测试命令或 [ 命令。
  # ! 是逻辑非运算符，用于反转条件的结果。
  # -d 是一个测试选项，用于检查 $D 是否是一个目录。
  # $D 是一个变量，代表目录的路径。

  if echo $1|grep ,  &>/dev/null;then   # &>重定向，&>/dev/null: 重定向 grep 命令的标准输出和错误输出到 /dev/null，这样无论是否有匹配的输出都不会显示在屏幕上
    mytemp="`echo $mytemp|tr ',' '\n'`"  #  使用 tr 命令将逗号替换为换行符，推荐用法 mytemp= $(echo "1.log,2,data" | tr ',' '\n')
  fi  # if的结束，elif

  for i in $mytemp
  do
    if ! ls -l $i &>/dev/null;then
      continue
    fi

    if ls -l $i|grep -Ew 'etc|usr|bin' &>/dev/null;then
      echo "you do not delete [ $i ]"
      continue
    fi

    # grep -Ew 'etc|usr|bin': 使用 grep 命令查找包含 "etc", "usr", 或 "bin" 的行。
    #     -E 选项启用扩展正则表达式。
    #     -w 选项确保匹配完整的单词。


    echo "now you want to delete >>> $i <<<" && ls -l $i
    Help
    /bin/mv "$i" $D
    echo "`date +%F-%T` --- "$i" Moved to $D" >> /root/mv_data.log
  done

  if [[ `ls -1 $D|wc -l` -eq 0 ]];then
    rm -r $D
  fi
}

main $1


```
