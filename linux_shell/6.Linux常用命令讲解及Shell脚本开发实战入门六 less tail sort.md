tail      倒序查看、持续监听文件最新的变化内容    tail -10 init.sh 查日志
tail -f 1.log  监听文件改动
for i in `seq 10`; do echo $i>>1.log;sleep 1; done

计数及排序命令
wc     # cat -n 1.log|wc -l  行数计数
sort  #  排序 cat 1.log|sort   cat 1.log|sort -rn 倒序排
-r 选项告诉 sort 命令进行反向排序（即从大到小）。
-n 选项告诉 sort 命令按照数字进行排序（而非默认的字母顺序）。


uniq  去重   # cat 1.log|sort|uniq   cat 1.log|sort|uniq -c 会得到 重复次数   cat 1.log|sort|uniq -c|sort -k 1 -rn
