# 介绍

nfs 卷能将 NFS (网络文件系统) 挂载到你的 Pod 中。 不像 emptyDir 那样会在删除 Pod 的同时也会被删除，nfs 卷的内容在删除 Pod 时会被保存，卷只是被卸载。 这意味着 nfs 卷可以被预先填充数据，并且这些数据可以在 Pod 之间共享。

缺点是，这种经过网络访问的，不能保证非常稳定性，虽然是内网访问，
不能用于频繁读写，例如 mysql

# 每一个节点都执行一下以下命令

## 安装 nfs

yum install nfs-utils -y

## 启动 nfs

systemctl start nfs-server

## 查看 nfs 版本

cat /proc/fs/nfsd/versions

## 创建共享目录 Node1 节点中

df -h

mkdir -p /home/nfs
cd /home/nfs
mkdir rw
mkdir ro

## 设置共享目录 export

vim /etc/exports

/home/nfs/rw 192.168.1.0/24(rw,sync,no_subtree_check,no_root_squash)
/home/nfs/ro 192.168.1.0/24(ro,sync,no_subtree_check,no_root_squash)

## 重新加载

exportfs -f
systemctl reload nfs-server

## 到其他测试节 Mastert 点安装 nfs-utils 并加载测试

mkdir -p /mnt/nfs/rw
mkdir -p /mnt/nfs/ro

mount -t nfs 192.168.1.11:/home/nfs/rw /mnt/nfs/rw
mount -t nfs 192.168.1.11:/home/nfs/ro /mnt/nfs/ro

cd /mnt/nfs
cd ro/
ls

[root@k8s-master ro]# touch master
touch: cannot touch ‘master’: Read-only file system

使用 NFS

nfs-test-pd.yaml

```
apiVersion: v1
kind: Pod
metadata:
  name: nfs-test-pd1
spec:
  containers:
  - image: nginx
    name: test-container
    volumeMounts:
    - mountPath: /usr/share/nginx/html
      name: test-volume
  volumes:
  - name: test-volume
    nfs:
      server: 192.168.1.11 # 网络存储服务地址
      #path: /home/nfs/rw/www/wolfcode # 网络存储路径
      path: /home/nfs/rw/www/wolfcode # 网络存储路径
      readOnly: false # 是否只读

```

# 命令

```
mkdir -p /home/nfs/rw/www/wolfcode
kubectl create -f nfs-test-pd.yaml

kubectl get po -owide

curl 10.244.36.110

echo '<h1>heloworld</h1>'>> /home/nfs/rw/www/wolfcode/index.html

curl 10.244.36.110



删除 POD 不会影响， node1 中数据的存储
kubectl delete po nfs-test-pd1 nfs-test-pd2
```
