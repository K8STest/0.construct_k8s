# 介绍

nfs 卷能将 NFS (网络文件系统) 挂载到你的 Pod 中。 不像 emptyDir 那样会在删除 Pod 的同时也会被删除，nfs 卷的内容在删除 Pod 时会被保存，卷只是被卸载。 这意味着 nfs 卷可以被预先填充数据，并且这些数据可以在 Pod 之间共享。

```
# 注意，这个nfs要在每一个节点都安装才可以，这一点在新家node节点的时候需要注意

# 安装 nfs
yum install nfs-utils -y

# 启动 nfs
systemctl start nfs-server

# 查看 nfs 版本
cat /proc/fs/nfsd/versions

# 创建共享目录
mkdir -p /data/nfs
cd /data/nfs
mkdir rw
mkdir ro

mkdir -p /home/nfs
cd /home/nfs
mkdir rw
mkdir ro

# 设置共享目录 export ， node1节点汇总操作
vim /etc/exports

/home/nfs/rw 192.168.1.0/24(rw,sync,no_subtree_check,no_root_squash)
/home/nfs/ro 192.168.1.0/24(ro,sync,no_subtree_check,no_root_squash)
/data/nfs/rw 192.168.1.0/24(rw,sync,no_subtree_check,no_root_squash)
/data/nfs/ro 192.168.1.0/24(ro,sync,no_subtree_check,no_root_squash)
# 重新加载
exportfs -f
systemctl reload nfs-server


# 做测试，到其他测试节点安装 nfs-utils 并加载测试， master节点中做操作
mkdir -p /mnt/nfs/rw
mount -t nfs 192.168.1.11:/home/nfs/rw /mnt/nfs/rw
mount -t nfs 192.168.1.11:/home/nfs/ro /mnt/nfs/ro


```

# NFS 挂载到容器中

```

apiVersion: v1
kind: Pod
metadata:
  name: nfs-test-pd1
spec:
  containers:
  - image: nginx
    name: test-container
    volumeMounts:
    - mountPath: /usr/share/nginx/html
      name: test-volume
  volumes:
  - name: test-volume
    nfs:
      server: 192.168.1.11 # 网络存储服务地址
      path: /home/nfs/rw/www/wolfcode # 网络存储路径
      readOnly: false # 是否只读

```
