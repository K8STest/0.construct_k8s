# 创建 harbor 访问账号密码（需要将下访问的配置信息改成你自己的）

kubectl create secret docker-registry harbor-secret --docker-server=192.168.1.12:8858 --docker-username=admin --docker-password=Harbor12345 -n kube-devops

docker login -uadmin 192.168.1.12:8858

systemctl daemon-reload
systemctl restart docker

cd /opt/k8s/harbor/harbor # 替换为你的 Harbor 安装目录
sudo docker-compose down
sudo docker-compose up -d
